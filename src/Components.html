/** * Shared presentation helpers. These functions do not read sheets, * inspect permissions, or call Apps Script server functions. */ let activeDialogInvoker = null; let activeDialogOnClose = null; let activeSidePanelInvoker = null; let activeSidePanelOnClose = null; let componentKeyboardHandlerBound = false; function initialiseBaseComponents() { if (componentKeyboardHandlerBound) { return; } document.addEventListener('keydown', event => { if (event.key === 'Tab') { const activeHost = isDialogOpen() ? document.getElementById('dialog-host') : isSidePanelOpen() ? document.getElementById('side-panel-host') : null; if (activeHost) { trapOverlayFocus(event, activeHost); } return; } if (event.key !== 'Escape') { return; } if (isDialogOpen()) { closeDialog('escape'); return; } if (isSidePanelOpen()) { closeSidePanel('escape'); } }); componentKeyboardHandlerBound = true; } function trapOverlayFocus(event, host) { const focusable = Array.from(host.querySelectorAll( 'button:not([disabled]), [href], input:not([disabled]), ' + 'select:not([disabled]), textarea:not([disabled]), ' + '[tabindex]:not([tabindex="-1"])' )).filter(element => element.offsetParent !== null && element.tabIndex >= 0 ); if (focusable.length === 0) { event.preventDefault(); return; } const first = focusable[0]; const last = focusable[focusable.length - 1]; if (event.shiftKey && document.activeElement === first) { event.preventDefault(); last.focus(); } else if ( !event.shiftKey && document.activeElement === last ) { event.preventDefault(); first.focus(); } } function buildPageHeader(options) { const settings = options || {}; const title = String(settings.title || ''); const supportingText = String( settings.supportingText || '' ); return `

${escapeHtml(title)}

${supportingText ? `

${escapeHtml(supportingText)}

` : ''}
${settings.actionsHtml ? `
${settings.actionsHtml}
` : ''}
`; } function buildStatusBadge(label, tone) { return ` ${escapeHtml(label)} `; } function buildSummaryMetric(value, label, tone) { return `
${escapeHtml(value)} ${escapeHtml(label)}
`; } function buildLoadingState(message) { return `
${escapeHtml(message)}
`; } function buildEmptyState(options) { const settings = options || {}; return `
${settings.title ? `

${escapeHtml(settings.title)}

` : ''}

${escapeHtml(settings.message || '')}

${settings.actionsHtml || ''}
`; } function buildErrorState(options) { const settings = options || {}; return `
${settings.title ? `

${escapeHtml(settings.title)}

` : ''}

${escapeHtml(settings.message || '')}

${settings.actionsHtml || ''}
`; } function buildFilterBar(options) { const settings = options || {}; return `
${settings.fieldsHtml || ''}
${settings.actionsHtml ? `
${settings.actionsHtml}
` : ''}
`; } function buildRecordRow(options) { const settings = options || {}; return `

${escapeHtml(settings.title || '')}

${settings.metaHtml || ''}
${settings.statusHtml || ''} ${settings.actionsHtml ? `
${settings.actionsHtml}
` : ''}
`; } function openDialog(options) { const settings = options || {}; const host = document.getElementById('dialog-host'); if (!host) { return; } activeDialogInvoker = settings.invoker || document.activeElement; activeDialogOnClose = settings.onClose || null; host.hidden = false; host.innerHTML = `

${escapeHtml( settings.title || '' )}

${settings.contentHtml || ''}
${settings.actionsHtml ? `
${settings.actionsHtml}
` : ''}
`; document.body.classList.add('overlay-open'); host.querySelectorAll('[data-dialog-close]').forEach( control => { control.addEventListener('click', () => { closeDialog('close'); }); } ); const dialog = host.querySelector('.dialog-component'); const focusTarget = dialog ? dialog.querySelector( 'button, [href], input, select, textarea, ' + '[tabindex]:not([tabindex="-1"])' ) : null; if (focusTarget) { focusTarget.focus(); } } function closeDialog(result, restoreFocus) { const host = document.getElementById('dialog-host'); if (!host || host.hidden) { return; } const invoker = activeDialogInvoker; const onClose = activeDialogOnClose; host.hidden = true; host.innerHTML = ''; activeDialogInvoker = null; activeDialogOnClose = null; updateOverlayBodyState(); if (typeof onClose === 'function') { onClose(result); } if ( restoreFocus !== false && invoker && document.contains(invoker) ) { invoker.focus(); } } function isDialogOpen() { const host = document.getElementById('dialog-host'); return Boolean(host && !host.hidden); } function openSidePanel(options) { const settings = options || {}; const host = document.getElementById('side-panel-host'); if (!host) { return; } activeSidePanelInvoker = settings.invoker || document.activeElement; activeSidePanelOnClose = settings.onClose || null; host.hidden = false; host.innerHTML = `

${escapeHtml( settings.title || '' )}

${settings.contentHtml || ''}
`; document.body.classList.add('overlay-open'); host.querySelectorAll('[data-side-panel-close]').forEach( control => { control.addEventListener('click', () => { closeSidePanel('close'); }); } ); const panel = host.querySelector('.side-panel-component'); const focusTarget = panel ? panel.querySelector( '[data-side-panel-initial-focus], button, [href], ' + 'input, select, textarea' ) : null; if (focusTarget) { focusTarget.focus(); } } function closeSidePanel(result, restoreFocus) { const host = document.getElementById('side-panel-host'); if (!host || host.hidden) { return; } const invoker = activeSidePanelInvoker; const onClose = activeSidePanelOnClose; host.hidden = true; host.innerHTML = ''; activeSidePanelInvoker = null; activeSidePanelOnClose = null; updateOverlayBodyState(); if (typeof onClose === 'function') { onClose(result); } if ( restoreFocus !== false && invoker && document.contains(invoker) ) { invoker.focus(); } } function isSidePanelOpen() { const host = document.getElementById('side-panel-host'); return Boolean(host && !host.hidden); } function updateOverlayBodyState() { document.body.classList.toggle( 'overlay-open', isDialogOpen() || isSidePanelOpen() ); }
