let currentDashboardSessionId = null; let dashboardTodaySessions = []; let dashboardTodayDateValue = ''; let dashboardTodayRequestToken = 0; let dashboardSessionRequestToken = 0; function renderTodayDashboard() { const content = document.getElementById('content'); content.className = ''; content.innerHTML = buildLoadingState( t('dashboard.loadingToday') ); const requestToken = ++dashboardTodayRequestToken; google.script.run .withSuccessHandler(data => { if ( requestToken !== dashboardTodayRequestToken || currentRouteId !== 'today' ) { return; } renderDashboardData(data); }) .withFailureHandler(error => { if ( requestToken !== dashboardTodayRequestToken || currentRouteId !== 'today' ) { return; } renderDashboardError(error); }) .getDashboardData(); } function renderDashboardData(data) { dashboardTodaySessions = data.sessions || []; dashboardTodayDateValue = data.todayDateValue || data.todayLabel || ''; if (dashboardTodaySessions.length === 0) { currentDashboardSessionId = null; renderEmptyTodayDashboard(); return; } const retainedSession = dashboardTodaySessions.find(session => session.sessionId === currentDashboardSessionId ); const selectedData = data.selectedSession; if ( retainedSession && selectedData && selectedData.session.sessionId !== retainedSession.sessionId ) { renderTodaySessionLoading(retainedSession); loadDashboardSession(retainedSession.sessionId); return; } renderDashboardSession(selectedData); } function renderEmptyTodayDashboard() { const content = document.getElementById('content'); content.innerHTML = ` ${buildPageHeader({ title: t('dashboard.today.title'), supportingText: formatDateOnly(dashboardTodayDateValue), actionsHtml: buildTodayRefreshButton() })} ${buildEmptyState({ title: t('dashboard.noSessionToday'), message: t('dashboard.noSessionTodayHelp') })} ${buildVisitorScheduleDashboardLink()} `; initialiseTodayWorkspaceEvents(); } function renderTodaySessionLoading(session) { const content = document.getElementById('content'); const timeText = buildTimeText( session.startTime, session.endTime ); content.innerHTML = ` ${buildPageHeader({ title: session.title, supportingText: [ formatDateOnly( session.dateValue || dashboardTodayDateValue ), timeText ].filter(Boolean).join(' · '), actionsHtml: buildTodayHeaderActions(session.sessionId) })} ${buildLoadingState(t('dashboard.loadingSession'))} `; initialiseTodayWorkspaceEvents(); } function loadDashboardSession(sessionId) { const requestToken = ++dashboardSessionRequestToken; google.script.run .withSuccessHandler(data => { if ( requestToken !== dashboardSessionRequestToken || currentRouteId !== 'today' || currentDashboardSessionId !== sessionId ) { return; } renderDashboardSession(data); }) .withFailureHandler(error => { if ( requestToken !== dashboardSessionRequestToken || currentRouteId !== 'today' ) { return; } renderDashboardError(error); }) .getDashboardSession(sessionId); } function renderDashboardSession(data) { const content = document.getElementById('content'); if (!content || !data) { return; } const session = data.session; const summary = data.summary; const groups = data.groups || {}; const members = data.members || []; const timeText = buildTimeText( session.startTime, session.endTime ); currentDashboardSessionId = session.sessionId; content.innerHTML = ` ${buildPageHeader({ title: session.title, supportingText: [ formatDateOnly( session.dateValue || session.date || dashboardTodayDateValue ), timeText ].filter(Boolean).join(' · '), actionsHtml: buildTodayHeaderActions(session.sessionId) })} ${session.notes ? `
${escapeHtml( session.notes )}
` : ''} ${buildTodayPriorityArea(groups)}

${t('dashboard.actionSummary.title')}

${buildSummaryMetric( (groups.conflicts || []).length, t('dashboard.summary.conflicts'), (groups.conflicts || []).length > 0 ? 'danger' : 'neutral' )} ${buildSummaryMetric( summary.unavailable, t('dashboard.summary.unavailable'), summary.unavailable > 0 ? 'danger' : 'neutral' )} ${buildSummaryMetric( summary.noResponse, t('dashboard.summary.noResponse'), summary.noResponse > 0 ? 'warning' : 'neutral' )} ${buildSummaryMetric( summary.notYetRecorded, t('dashboard.summary.attendanceMissing'), summary.notYetRecorded > 0 ? 'warning' : 'neutral' )} ${buildSummaryMetric( summary.volunteers, t('dashboard.summary.volunteers'), 'information' )}
${t('dashboard.summary.activeMembersWithCount', { count: escapeHtml(summary.activeMembers) })}
${buildTodayVisitors(groups.volunteers || [])} ${buildTodayMemberStatus(members, groups)}
${buildTodayAttendanceSummary(summary)} ${buildVisitorScheduleDashboardLink()}
`; initialiseTodayWorkspaceEvents(); initialiseDisclosures(content); } function buildTodayHeaderActions(selectedSessionId) { const selector = dashboardTodaySessions.length > 1 ? ` ` : ''; return `${selector}${buildTodayRefreshButton()}`; } function buildTodayRefreshButton() { return ` `; } function initialiseTodayWorkspaceEvents() { const selector = document.getElementById( 'dashboard-session-selector' ); const refreshButton = document.getElementById( 'refresh-dashboard-button' ); const visitorButton = document.getElementById( 'open-visitor-schedules' ); if (selector) { selector.addEventListener('change', () => { currentDashboardSessionId = selector.value; const selectedSession = dashboardTodaySessions.find( session => session.sessionId === selector.value ); if (selectedSession) { renderTodaySessionLoading(selectedSession); } loadDashboardSession(selector.value); }); } if (refreshButton) { refreshButton.addEventListener('click', () => { refreshButton.disabled = true; refreshButton.textContent = t('common.refreshing'); if (currentDashboardSessionId) { loadDashboardSession(currentDashboardSessionId); } else { renderTodayDashboard(); } }); } if (visitorButton) { visitorButton.addEventListener('click', () => { navigateToRoute('visitor-coordination'); }); } } function buildTodayPriorityArea(groups) { const conflicts = groups.conflicts || []; const attendanceMissing = groups.attendanceMissing || []; const banners = []; if (conflicts.length > 0) { banners.push(buildPriorityBanner({ tone: 'danger', title: t('dashboard.attention.conflictsTitle'), message: t('dashboard.attention.conflictsMessage', { count: conflicts.length }), detailsHtml: buildTodayPriorityNames(conflicts) })); } if (attendanceMissing.length > 0) { banners.push(buildPriorityBanner({ tone: 'warning', title: t('dashboard.attention.attendanceTitle'), message: t('dashboard.attention.attendanceMessage', { count: attendanceMissing.length }), detailsHtml: buildTodayPriorityNames(attendanceMissing) })); } if (banners.length === 0) { return ''; } return `

${t('dashboard.attention.title')}

${banners.join('')}
`; } function buildTodayPriorityNames(members) { return `
    ${members.map(member => `
  • ${escapeHtml(member.name)}
    • `).join('')}
`; } function buildTodayVisitors(visitors) { const rows = visitors.map(member => { const volunteer = member.volunteer || {}; const details = [ member.grade, volunteer.activity, volunteer.location, volunteer.departureTime ? t('dashboard.volunteer.departureTime', { time: formatTime(volunteer.departureTime) }) : '' ].filter(Boolean); const isConflict = member.availability === 'Unavailable'; return buildRecordRow({ title: member.name, metaHtml: `

${details.map(escapeHtml).join(' · ')}

${member.reason ? `

${escapeHtml( member.reason )}

` : ''} `, statusHtml: `
${buildStatusBadge( getDashboardAvailabilityLabel(member.availability), getAvailabilityTone(member.availability) )} ${isConflict ? buildStatusBadge( t('dashboard.summary.conflicts'), 'danger' ) : ''}
` }); }).join(''); return `

${t('dashboard.visitors.todayTitle')}

${buildStatusBadge( String(visitors.length), visitors.length > 0 ? 'information' : 'neutral' )}
${visitors.length > 0 ? `
${rows}
` : buildEmptyState({ message: t('dashboard.visitors.noneToday') })}
`; } function buildTodayMemberStatus(members, groups) { const availabilityGroups = [ { id: 'today-available', title: t('dashboard.group.available'), members: members.filter(member => member.availability === 'Available' ), type: 'availability' }, { id: 'today-unavailable', title: t('dashboard.group.unavailable'), members: members.filter(member => member.availability === 'Unavailable' ), type: 'availability' }, { id: 'today-unsure', title: t('dashboard.group.unsure'), members: members.filter(member => member.availability === 'Unsure' ), type: 'availability' }, { id: 'today-no-response', title: t('dashboard.group.noResponse'), members: members.filter(member => !member.availability), type: 'availability' } ]; const attendanceGroups = [ { id: 'today-attendance-recorded', title: t('dashboard.group.attendanceRecorded'), members: groups.attendanceRecorded || [], type: 'attendance' }, { id: 'today-attendance-missing', title: t('dashboard.group.attendanceMissing'), members: groups.attendanceMissing || [], type: 'attendance' } ]; return `

${t('dashboard.memberStatus.title')}

${t('dashboard.memberStatus.availability')}

${availabilityGroups.map((group, index) => buildTodayMemberDisclosure(group, index === 0) ).join('')}

${attendanceGroups.map(group => buildTodayMemberDisclosure(group, false) ).join('')}
`; } function buildTodayMemberDisclosure(group, expanded) { const list = group.members || []; return buildDisclosure({ id: group.id, title: group.title, count: list.length, expanded: expanded, contentHtml: list.length > 0 ? `
    ${list.map(member => buildTodayMember(member, group.type) ).join('')}
` : `
${t( 'dashboard.group.empty' )}
` }); } function buildTodayMember(member, context) { const details = [member.grade]; if (context === 'availability') { details.push(getDashboardAvailabilityLabel( member.availability )); } else if (member.attendance) { details.push(t('dashboard.attendance.recorded')); } else { details.push(t('dashboard.attendance.notRecorded')); } return `
  • ${escapeHtml(member.name)}
    ${details.filter(Boolean).map(escapeHtml).join(' · ')}
    ${context === 'availability' && member.reason ? `
    ${escapeHtml( member.reason )}
    ` : ''}
    • `; } function buildTodayAttendanceSummary(summary) { return `

    ${t('dashboard.attendance.title')}

    ${buildStatusBadge( t('dashboard.attendance.readOnly'), 'neutral' )}

    ${t('dashboard.attendance.summary', { recorded: `${escapeHtml( summary.attendanceRecorded )}`, expected: `${escapeHtml( summary.expected )}` })}

    ${t('dashboard.attendance.futureNotice')}

    `; } function buildVisitorScheduleDashboardLink() { return `

    ${t('dashboard.visitors.title')}

    ${t('dashboard.visitors.description')}

    `; } function getDashboardAvailabilityLabel(response) { const keys = { Available: 'availability.status.available', Unavailable: 'availability.status.unavailable', Unsure: 'availability.status.unsure' }; return t(keys[response] || 'availability.status.noResponse'); } function getAvailabilityTone(response) { const tones = { Available: 'success', Unavailable: 'danger', Unsure: 'warning' }; return tones[response] || 'neutral'; } function renderDashboardError() { const content = document.getElementById('content'); if (!content) { return; } content.className = ''; content.innerHTML = ` ${buildPageHeader({ title: t('dashboard.today.title'), actionsHtml: buildTodayRefreshButton() })} ${buildErrorState({ title: t('common.error'), message: t('dashboard.error.loadFailed') })} `; initialiseTodayWorkspaceEvents(); }
