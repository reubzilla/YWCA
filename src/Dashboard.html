let currentDashboardSessionId = null; let currentDashboardSection = 'today'; let dashboardTodayRequestToken = 0; let dashboardSessionRequestToken = 0; function renderDashboard() { const content = document.getElementById('content'); content.className = ''; content.innerHTML = `

${t('dashboard.title')}

`; initialiseDashboardSectionNavigation(); renderActiveDashboardSection(); } function initialiseDashboardSectionNavigation() { document .querySelectorAll('[data-dashboard-section]') .forEach(button => { button.addEventListener('click', () => { navigateToRoute( button.dataset.dashboardSection === 'upcoming' ? 'planning' : 'today' ); }); }); updateDashboardSectionNavigation(); } function updateDashboardSectionNavigation() { document .querySelectorAll('[data-dashboard-section]') .forEach(button => { const isActive = button.dataset.dashboardSection === currentDashboardSection; button.classList.toggle('active', isActive); button.setAttribute( 'aria-selected', String(isActive) ); }); } function renderActiveDashboardSection() { if (currentDashboardSection === 'upcoming') { renderUpcomingActivities(); return; } renderTodayDashboard(); } function renderTodayDashboard() { const content = document.getElementById( 'dashboard-section-content' ); content.innerHTML = `
${t('dashboard.loadingToday')}
`; const requestToken = ++dashboardTodayRequestToken; google.script.run .withSuccessHandler(data => { if ( requestToken !== dashboardTodayRequestToken || currentView !== 'dashboard' || currentDashboardSection !== 'today' ) { return; } renderDashboardData(data); }) .withFailureHandler(error => { if ( requestToken !== dashboardTodayRequestToken || currentView !== 'dashboard' || currentDashboardSection !== 'today' ) { return; } renderDashboardError(error); }) .getDashboardData(); } function renderDashboardData(data) { const content = document.getElementById('dashboard-section-content'); const sessions = data.sessions || []; if (sessions.length === 0) { content.innerHTML = `

${escapeHtml(formatDateOnly( data.todayDateValue || data.todayLabel ))}

${t('dashboard.noSessionToday')}
${buildVisitorScheduleDashboardLink()} `; initialiseVisitorScheduleDashboardLink(); return; } let selector = ''; if (sessions.length > 1) { selector = ` `; } content.innerHTML = `

${escapeHtml(formatDateOnly( data.todayDateValue || data.todayLabel ))}

${selector}
${buildVisitorScheduleDashboardLink()}
`; initialiseVisitorScheduleDashboardLink(); if (sessions.length > 1) { const selectorElement = document.getElementById( 'dashboard-session-selector' ); const retainedSession = sessions.find(session => session.sessionId === currentDashboardSessionId ); if (retainedSession) { selectorElement.value = retainedSession.sessionId; } selectorElement.addEventListener( 'change', () => { currentDashboardSessionId = selectorElement.value; loadDashboardSession( selectorElement.value ); } ); } const selectedSession = sessions.find(session => session.sessionId === currentDashboardSessionId ); if ( selectedSession && data.selectedSession && data.selectedSession.session.sessionId !== selectedSession.sessionId ) { loadDashboardSession(selectedSession.sessionId); } else { renderDashboardSession(data.selectedSession); } } function buildVisitorScheduleDashboardLink() { return `

${t('dashboard.visitors.title')}

${t('dashboard.visitors.description')}

`; } function initialiseVisitorScheduleDashboardLink() { const button = document.getElementById( 'open-visitor-schedules' ); if (button) { button.addEventListener('click', () => { selectView('volunteer'); }); } } function loadDashboardSession(sessionId) { const container = document.getElementById( 'dashboard-session-content' ); container.innerHTML = `
${t('dashboard.loadingSession')}
`; const requestToken = ++dashboardSessionRequestToken; google.script.run .withSuccessHandler(data => { if ( requestToken !== dashboardSessionRequestToken || currentView !== 'dashboard' || currentDashboardSection !== 'today' ) { return; } renderDashboardSession(data); }) .withFailureHandler(error => { if ( requestToken !== dashboardSessionRequestToken || currentView !== 'dashboard' || currentDashboardSection !== 'today' ) { return; } renderDashboardError(error); }) .getDashboardSession(sessionId); } function renderDashboardSession(data) { const container = document.getElementById( 'dashboard-session-content' ); if (!container || !data) { return; } currentDashboardSessionId = data.session.sessionId; const session = data.session; const summary = data.summary; const groups = data.groups; const timeText = buildTimeText( session.startTime, session.endTime ); let html = `

${escapeHtml(session.title)}

${escapeHtml(formatDateOnly( session.dateValue || session.date ))} ${timeText ? `
${escapeHtml(timeText)}` : ''}

${session.notes ? `
${escapeHtml(session.notes)}
` : ''}
${buildSummaryCard( summary.activeMembers, t('dashboard.summary.activeMembers'), '' )} ${buildSummaryCard( summary.available, t('dashboard.summary.available'), 'positive' )} ${buildSummaryCard( summary.volunteers, t('dashboard.summary.volunteers'), 'positive' )} ${buildSummaryCard( summary.unavailable, t('dashboard.summary.unavailable'), summary.unavailable > 0 ? 'problem' : '' )} ${buildSummaryCard( summary.unsure, t('dashboard.summary.unsure'), summary.unsure > 0 ? 'attention' : '' )} ${buildSummaryCard( summary.noResponse, t('dashboard.summary.noResponse'), summary.noResponse > 0 ? 'attention' : '' )}
`; if ( groups.conflicts && groups.conflicts.length > 0 ) { html += buildConflictBox(groups.conflicts); } html += buildDashboardGroup( t('dashboard.group.volunteers'), groups.volunteers, 'volunteer' ); html += buildDashboardGroup( t('dashboard.group.available'), groups.available, 'available' ); html += buildDashboardGroup( t('dashboard.group.unavailable'), groups.unavailable, 'unavailable' ); html += buildDashboardGroup( t('dashboard.group.unsure'), groups.unsure, 'unsure' ); html += buildDashboardGroup( t('dashboard.group.noResponse'), groups.noResponse, 'no-response' ); html += `

${t('dashboard.attendance.title')}

${t('dashboard.attendance.summary', { recorded: `${summary.attendanceRecorded}`, expected: `${summary.expected}` })}

${t('dashboard.attendance.futureNotice')}

`; container.innerHTML = html; const refreshButton = document.getElementById( 'refresh-dashboard-button' ); if (refreshButton) { refreshButton.addEventListener('click', () => { refreshButton.disabled = true; refreshButton.textContent = t('common.refreshing'); loadDashboardSession(session.sessionId); }); } } function buildSummaryCard( number, label, className ) { return `
${escapeHtml(number)} ${escapeHtml(label)}
`; } function buildDashboardGroup( title, members, type ) { const list = members || []; return `

${escapeHtml(title)}

${list.length}
${list.length === 0 ? `
${t('dashboard.group.empty')}
` : `
    ${list.map(member => buildDashboardMember(member, type) ).join('')}
`}
`; } function buildDashboardMember(member, type) { let details = []; if (member.grade) { details.push(member.grade); } if (type === 'volunteer' && member.volunteer) { if (member.volunteer.activity) { details.push(member.volunteer.activity); } if (member.volunteer.location) { details.push(member.volunteer.location); } if (member.volunteer.departureTime) { details.push( t('dashboard.volunteer.departureTime', { time: formatTime( member.volunteer.departureTime ) }) ); } } return `
  • ${escapeHtml(member.name)}
    ${details.length > 0 ? `
    ${details .map(escapeHtml) .join(' · ')}
    ` : ''} ${member.reason ? `
    ${escapeHtml(member.reason)}
    ` : ''}
    • `; } function buildConflictBox(conflicts) { return `

    ${t('dashboard.conflict.title')}

    ${t('dashboard.conflict.description')}

      ${conflicts.map(member => `
    • ${escapeHtml(member.name)} ${member.reason ? ` — ${escapeHtml(member.reason)}` : ''}
      • `).join('')}
    `; } function renderDashboardError(error) { const content = document.getElementById('dashboard-section-content'); if (!content) { return; } content.className = ''; content.innerHTML = `
    ${escapeHtml(t('dashboard.error.loadFailed'))}
    `; }
