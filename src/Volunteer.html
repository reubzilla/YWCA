${t('volunteer.intro')}

` : assignments.map(buildVolunteerAssignmentCard).join('')} `; } function buildVolunteerAssignmentCard(assignment) { const date = assignment.dateValue ? formatDateOnly(assignment.dateValue) : t('volunteer.sessionUnavailable'); const title = assignment.sessionTitle || t('volunteer.sessionUnavailable'); return `
${escapeHtml(date)}

${escapeHtml(title)}

${escapeHtml(translateBusinessValue( 'assignmentStatus', assignment.assignmentStatus ))}

${escapeHtml( assignment.location || t('volunteer.notSet') )}
${t('volunteer.departureTime')}
${escapeHtml( assignment.departureTime ? formatTime(assignment.departureTime) : t('volunteer.notSet') )}
${assignment.notes ? `
