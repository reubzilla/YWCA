let personalAssignments = []; let personalAssignmentsToday = ''; let personalAssignmentRequestToken = 0; function renderVolunteer() { const content = document.getElementById('content'); content.className = ''; content.innerHTML = `${buildPageHeader({ title: t('volunteer.title'), supportingText: t('volunteer.intro') })}${buildLoadingState(t('volunteer.loading'))}`; const token = ++personalAssignmentRequestToken; google.script.run .withSuccessHandler(data => { if (token !== personalAssignmentRequestToken || currentRouteId !== 'assignments') return; renderVolunteerAssignments(data); }) .withFailureHandler(() => { if (token === personalAssignmentRequestToken) renderVolunteerError(); }) .getMyVolunteerAssignments(); } function renderVolunteerAssignments(data) { personalAssignments = data.assignments || []; personalAssignmentsToday = data.todayDateValue || getTokyoTodayDateValue(); const target = consumePersonalRouteTarget('assignments'); const targetSessionId = target?.sessionId || ''; const groups = groupPersonalAssignments(personalAssignments, personalAssignmentsToday); const content = document.getElementById('content'); content.innerHTML = ` ${buildPageHeader({title: t('volunteer.title'), supportingText: t('volunteer.intro')})} ${!personalAssignments.length ? buildEmptyState({message: t('volunteer.empty')}) : ` ${groups.next ? `

${t('volunteer.nextAssignment')}

${buildAssignmentCard(groups.next, { prominent: true, id: `personal-assignment-${groups.next.sessionId}` })}
` : ''} ${groups.future.length ? `

${t('volunteer.futureAssignments')}

${groups.future.map(assignment => buildAssignmentCard(assignment, { prominent: assignment.sessionId === targetSessionId, id: `personal-assignment-${assignment.sessionId}` }) ).join('')}
` : ''} ${groups.history.length ? `

${t('volunteer.history')}

${groups.history.map(assignment => buildAssignmentCard(assignment, { prominent: assignment.sessionId === targetSessionId, id: `personal-assignment-${assignment.sessionId}` }) ).join('')}
` : ''} `}`; if (targetSessionId) { window.requestAnimationFrame(() => document.getElementById( `personal-assignment-${targetSessionId}` )?.scrollIntoView({block: 'center'})); } } function groupPersonalAssignments(assignments, todayDateValue) { const active = assignments.filter(item => item.dateValue >= todayDateValue && !['Cancelled', 'Declined'].includes(item.assignmentStatus) ); const next = active[0] || null; return { next, future: next ? active.slice(1) : active, history: assignments.filter(item => item.dateValue < todayDateValue || ['Cancelled', 'Declined'].includes(item.assignmentStatus) ) }; } function renderVolunteerError() { const content = document.getElementById('content'); content.innerHTML = `${buildPageHeader({title: t('volunteer.title')})}${buildErrorState({ message: t('volunteer.error.loadFailed') })}`; }
