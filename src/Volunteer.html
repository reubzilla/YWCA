let personalAssignments = []; let personalAssignmentsToday = ''; let personalAssignmentRequestToken = 0; let personalAssignmentHistoryPage = { nextOffset: 0, hasMore: false, includesOlder: false }; let personalAssignmentHistoryLoading = false; let personalAssignmentHistoryErrorKey = ''; function renderVolunteer() { personalAssignmentHistoryLoading = false; personalAssignmentHistoryErrorKey = ''; personalAssignmentHistoryPage = { nextOffset: 0, hasMore: false, includesOlder: false }; const content = document.getElementById('content'); content.className = ''; content.innerHTML = `${buildPageHeader({ title: t('volunteer.title'), supportingText: t('volunteer.intro') })}${buildLoadingState(t('volunteer.loading'))}`; const token = ++personalAssignmentRequestToken; google.script.run .withSuccessHandler(data => { if (token !== personalAssignmentRequestToken || currentRouteId !== 'assignments') return; renderVolunteerAssignments(data); }) .withFailureHandler(() => { if (token === personalAssignmentRequestToken) renderVolunteerError(); }) .getMyVolunteerAssignments(); } function renderVolunteerAssignments(data) { personalAssignments = data.assignments || []; personalAssignmentsToday = data.todayDateValue || getTokyoTodayDateValue(); personalAssignmentHistoryPage = data.historyPage || personalAssignmentHistoryPage; const target = consumePersonalRouteTarget('assignments'); const targetSessionId = target?.sessionId || ''; const groups = groupPersonalAssignments(personalAssignments, personalAssignmentsToday); const content = document.getElementById('content'); content.innerHTML = ` ${buildPageHeader({title: t('volunteer.title'), supportingText: t('volunteer.intro')})} ${!personalAssignments.length ? buildEmptyState({message: t('volunteer.empty')}) : ` ${groups.today.length ? `

${t('volunteer.todayAssignment')}

${groups.today.map(assignment => buildAssignmentCard(assignment, { prominent: true, id: `personal-assignment-${assignment.sessionId}` }) ).join('')}
` : ''} ${groups.next ? `

${t('volunteer.nextAssignment')}

${buildAssignmentCard(groups.next, { prominent: true, id: `personal-assignment-${groups.next.sessionId}` })}
` : ''} ${groups.future.length ? `

${t('volunteer.futureAssignments')}

${groups.future.map(assignment => buildAssignmentCard(assignment, { prominent: assignment.sessionId === targetSessionId, id: `personal-assignment-${assignment.sessionId}` }) ).join('')}
` : ''} ${groups.history.length || personalAssignmentHistoryPage.hasMore ? `

${t('history.recent')}

${groups.history.length ? `

${escapeHtml(t( 'history.showingMostRecent', {count: groups.history.length} ))}

${groups.history.map(assignment => buildAssignmentCard(assignment, { prominent: assignment.sessionId === targetSessionId, id: `personal-assignment-${assignment.sessionId}` }) ).join('')}
` : ''} ${personalAssignmentHistoryErrorKey ? `

${escapeHtml(t( personalAssignmentHistoryErrorKey ))}

` : ''} ${personalAssignmentHistoryPage.hasMore ? `` : personalAssignmentHistoryPage.includesOlder ? `

${t('history.noOlderActivities')}

` : ''}
` : ''} `}`; document.getElementById('assignment-load-older')?.addEventListener( 'click', loadOlderPersonalAssignments ); if (targetSessionId) { window.requestAnimationFrame(() => document.getElementById( `personal-assignment-${targetSessionId}` )?.scrollIntoView({block: 'center'})); } } function loadOlderPersonalAssignments() { if ( personalAssignmentHistoryLoading || !personalAssignmentHistoryPage.hasMore ) { return; } personalAssignmentHistoryLoading = true; personalAssignmentHistoryErrorKey = ''; renderVolunteerAssignments({ assignments: personalAssignments, todayDateValue: personalAssignmentsToday, historyPage: personalAssignmentHistoryPage }); const requestToken = personalAssignmentRequestToken; const currentHistoryCount = groupPersonalAssignments( personalAssignments, personalAssignmentsToday ).history.length; google.script.run .withSuccessHandler(data => { if ( requestToken !== personalAssignmentRequestToken || currentRouteId !== 'assignments' ) { return; } const knownSessionIds = new Set( personalAssignments.map(item => item.sessionId) ); (data.assignments || []).forEach(item => { if (!knownSessionIds.has(item.sessionId)) { personalAssignments.push(item); } }); personalAssignmentHistoryPage = data.historyPage || { nextOffset: currentHistoryCount, hasMore: false, includesOlder: true }; personalAssignmentHistoryLoading = false; renderVolunteerAssignments({ assignments: personalAssignments, todayDateValue: personalAssignmentsToday, historyPage: personalAssignmentHistoryPage }); }) .withFailureHandler(() => { personalAssignmentHistoryLoading = false; personalAssignmentHistoryErrorKey = 'history.loadOlderFailed'; renderVolunteerAssignments({ assignments: personalAssignments, todayDateValue: personalAssignmentsToday, historyPage: personalAssignmentHistoryPage }); }) .getMyVolunteerAssignments({ includeOlder: true, historyOffset: currentHistoryCount }); } function groupPersonalAssignments(assignments, todayDateValue) { const active = assignments.filter(item => !['Cancelled', 'Declined'].includes(item.assignmentStatus) ); const today = active.filter(item => classifyDateOnlyValue(item.dateValue, todayDateValue) === DATE_CLASSIFICATIONS.TODAY ); const future = active.filter(item => classifyDateOnlyValue(item.dateValue, todayDateValue) === DATE_CLASSIFICATIONS.UPCOMING ); const next = future[0] || null; return { today, next, future: next ? future.slice(1) : future, history: assignments.filter(item => ['Cancelled', 'Declined'].includes(item.assignmentStatus) || ![ DATE_CLASSIFICATIONS.TODAY, DATE_CLASSIFICATIONS.UPCOMING ].includes(classifyDateOnlyValue(item.dateValue, todayDateValue)) ) }; } function renderVolunteerError() { const content = document.getElementById('content'); content.innerHTML = `${buildPageHeader({title: t('volunteer.title')})}${buildErrorState({ message: t('volunteer.error.loadFailed') })}`;
