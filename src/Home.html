function renderHome() { const content = document.getElementById('content'); const user = portalData.user; const notifications = portalData.notifications || []; const sessions = portalData.sessions || []; let html = `

${t('home.welcome', { name: escapeHtml(user.name) })}

${escapeHtml(translateBusinessValue( 'role', user.role ))} ${user.grade ? ` · ${escapeHtml(user.grade)}` : ''}
${escapeHtml(user.email)}

${t('home.notificationsTitle')}

`; if (notifications.length === 0) { html += `
${t('home.noNotifications')}
`; } else { notifications.forEach(notification => { html += `

${escapeHtml(notification.title)}

${escapeHtml(notification.message)}

${escapeHtml(formatDateOnly( notification.dateValue || notification.date ))}
`; }); } html += `

${t('home.upcomingSessionsTitle')}

`; if (sessions.length === 0) { html += `
${t('home.noUpcomingSessions')}
`; } else { sessions.forEach(session => { const timeText = buildTimeText( session.startTime, session.endTime ); html += `
${escapeHtml(formatDateOnly( session.dateValue || session.date ))}

${escapeHtml(session.title)}

${timeText ? escapeHtml(timeText) : t('session.timeNotSpecified')}

${escapeHtml(translateBusinessValue( 'sessionType', session.sessionType ))} ${session.responseDeadline ? `

${t('session.responseDeadlineLabel')} ${escapeHtml( formatDateTime(session.responseDeadline) )}

` : ''} ${session.notes ? `
${escapeHtml(session.notes)}
` : ''}
`; }); } content.className = ''; content.innerHTML = html; }
